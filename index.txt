async function callAPI(asPDF){
  const pdfExtra=asPDF?'\n- This is a full PDF — extract County from the document header on the first page':'\n- This is a screenshot — County may not be visible, return "—" if not found';
  const prompt=`You are a real estate deed parser. Return ONLY raw JSON, no markdown, no explanation.
RULES: Only extract what is visible. Use "—" for missing fields. Recorded Date is in the filing stamp upper corner. Map # may be Tax Map Number/Parcel ID/TMS.${pdfExtra}
Taxes Owed: look for delinquent tax amounts or tax liability near prior owner info.
Return ONLY: {"Document Type":"","Tax Sale Auction Date":"","Recorded Date":"","Seller":"","Sale Price":"","Grantee":"","Grantee Address":"","Prior Owner(s)":"","Taxes Owed":"","Property Address":"","County":"","Map #":"","Legal Description":""}`;
  const content=pageImages.map(b64=>({type:'image',source:{type:'base64',media_type:'image/png',data:b64}}));
  content.push({type:'text',text:prompt});
  const res=await fetch('https://api.anthropic.com/v1/messages',{method:'POST',headers:{'Content-Type':'application/json','x-api-key':apiKey,'anthropic-version':'2023-06-01','anthropic-dangerous-direct-browser-access':'true'},body:JSON.stringify({model:'claude-sonnet-4-20250514',max_tokens:2000,messages:[{role:'user',content}]})});
  const data=await res.json();
  if(data.error)throw new Error(data.error.message);
  let text=data.content.map(c=>c.text||'').join('').replace(/```json|```/g,'').trim();
  return JSON.parse(text);
}

function normalizeFields(parsed){
  const req=['Document Type','Tax Sale Auction Date','Recorded Date','Seller','Sale Price','Grantee','Grantee Address','Prior Owner(s)','Taxes Owed','Property Address','County','Map #','Legal Description'];
  req.forEach(f=>{if(!parsed[f]||parsed[f]==='null'||parsed[f]===null)parsed[f]='—';});
}

function checkDuplicate(parsed){
  const mapNum=parsed['Map #']!=='—'?parsed['Map #']:'';
  const addr=parsed['Property Address']!=='—'?parsed['Property Address']:'';
  if(!mapNum&&!addr)return null;
  return scanHistory.find(h=>{
    const hMap=h.data['Map #']!=='—'?h.data['Map #']:'';
    const hAddr=h.data['Property Address']!=='—'?h.data['Property Address']:'';
    if(mapNum&&hMap&&mapNum.replace(/[-. ]/g,'')===hMap.replace(/[-. ]/g,''))return true;
    if(addr&&hAddr&&addr.toLowerCase().trim()===hAddr.toLowerCase().trim())return true;
    return false;
  })||null;
}

function formatShortDate(str){
  if(!str||str==='—')return'';
  const m=str.match(/([A-Za-z]+)\s+(\d{1,2}),?\s+(\d{4})/);
  if(m)return m[1].slice(0,3)+' '+m[2]+', '+m[3];
  const d=new Date(str);
  if(!isNaN(d))return d.toLocaleDateString('en-US',{month:'short',day:'numeric',year:'numeric'});
  return str.split(' ')[0];
}

function row(k,v){
  const safe=String(v).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;').replace(/'/g,'&#39;');
  const isEmpty=v==='—'||!v;
  const fieldId='field-'+k.replace(/[^a-zA-Z0-9]/g,'_');
  const valHtml=isEmpty?`<span class="result-val" id="${fieldId}" data-field="${k}" style="color:#ccc;font-style:italic;font-weight:400;">not found in document</span>`:`<span class="result-val" id="${fieldId}" data-field="${k}">${safe}</span>`;
  let btnHtml=`<button class="copy-btn edit-btn" onclick="editField('${fieldId}','${k.replace(/'/g,"\\'")}')">Edit</button>`;
  if(!isEmpty){
    btnHtml+=`<button class="copy-btn" id="copy-${fieldId}" onclick="copyField(this,'${safe}')">Copy</button>`;
    if(k==='Map #'){
      const plain=String(v).replace(/[-. ]/g,'');
      const ps=plain.replace(/'/g,'&#39;').replace(/"/g,'&quot;');
      btnHtml+=`<button class="copy-btn" onclick="copyField(this,'${ps}')" title="No hyphens/periods"># Plain</button>`;
    }
  }
  return`<div class="result-row"><span class="result-key">${k}</span>${valHtml}<span style="display:flex;gap:3px;flex-shrink:0;align-items:flex-start">${btnHtml}</span></div>`;
}

function editField(fieldId,fieldKey){
  const span=document.getElementById(fieldId);if(!span)return;
  const current=span.dataset.rawVal||span.textContent;
  const isPlaceholder=current==='not found in document';
  const input=document.createElement('input');input.type='text';input.value=isPlaceholder?'':current;
  input.style.cssText='width:100%;background:#f0f4f8;border:1px solid #1a3a5c;border-radius:4px;padding:3px 7px;font-size:12px;font-family:inherit;color:#0f1f2e;outline:none;';
  span.replaceWith(input);input.focus();input.select();
  function save(){
    const nv=input.value.trim()||'—';
    const ns=document.createElement('span');ns.className='result-val';ns.id=fieldId;ns.dataset.field=fieldKey;ns.dataset.rawVal=nv;
    if(nv==='—'){ns.style.cssText='color:#ccc;font-style:italic;font-weight:400;';ns.textContent='not found in document';}else{ns.textContent=nv;}
    input.replaceWith(ns);
    if(currentData)currentData[fieldKey]=nv;
    const cb=document.getElementById('copy-'+fieldId);
    if(cb){const safeNew=nv.replace(/'/g,"&#39;").replace(/"/g,'&quot;');cb.onclick=function(){copyField(this,safeNew);};cb.style.display=nv==='—'?'none':'';}
  }
  input.addEventListener('blur',save);
  input.addEventListener('keydown',e=>{if(e.key==='Enter'){e.preventDefault();input.blur();}if(e.key==='Escape'){input.value=isPlaceholder?'':current;input.blur();}});
}

function renderHistoryPage(){
  const search=(document.getElementById('histSearch').value||'').toLowerCase();

  let items=[...scanHistory];
  const statusFilter=document.getElementById('histStatusFilter').value;
  if(countyFilter)items=items.filter(h=>h.meta.county===countyFilter);
  if(statusFilter)items=items.filter(h=>(h.meta.status||'No Action Yet')===statusFilter);
  if(search)items=items.filter(h=>{const hay=[h.meta.title,h.meta.county,h.data['Grantee'],h.data['Property Address'],h.data['Seller'],h.data['Map #'],h.data['Taxes Owed']].join(' ').toLowerCase();return hay.includes(search);});

    <div class="dash-card"><div class="dash-val">$${totalTax>=1000000?(totalTax/1000000).toFixed(1)+'M':totalTax>=1000?Math.round(totalTax/1000)+'K':Math.round(totalTax).toLocaleString()}</div><div class="dash-label">Total Taxes Owed</div></div>

    if (!colSortKey) return list;
    return [...list].sort((a, b) => {
      let va = a.data[colSortKey] || '—';
      let vb = b.data[colSortKey] || '—';
      // Try numeric sort for price/tax fields
      const na = parseFloat(va.replace(/[$,]/g,''));
      const nb = parseFloat(vb.replace(/[$,]/g,''));
      if (!isNaN(na) && !isNaN(nb)) {
        return colSortDir === 'asc' ? na - nb : nb - na;
      }
      // Date sort
      const da = new Date(va), db = new Date(vb);
      if (!isNaN(da) && !isNaN(db)) {
        return colSortDir === 'asc' ? da - db : db - da;
      }
      // String sort
      return colSortDir === 'asc' ? va.localeCompare(vb) : vb.localeCompare(va);
    });

  }

    const fg = group.filter(h => new Date(h.id).getFullYear() === activeYear);

    html+=`<div class="county-section" id="${cid}">

        <span class="county-section-name">${county} County</span>

    if(groupBy==='date'){

    }

  });

  body.innerHTML=html;
}

function restoreData(input){
  const file=input.files[0];
  if(!file)return;
  const reader=new FileReader();
  reader.onload=function(e){
    try{
      const backup=JSON.parse(e.target.result);
      if(!backup.history||!Array.isArray(backup.history)){
        alert('Invalid backup file. Please use a file exported from Tax Deed Scan.');
        return;
      }
      const incoming=backup.history;
      const existingIds=new Set(scanHistory.map(function(h){return h.id;}));
      const newRecords=incoming.filter(function(h){return !existingIds.has(h.id);});
      const duplicateCount=incoming.length-newRecords.length;
      if(incoming.length===0){alert('Backup file contains no records.');return;}
      let msg='Found '+incoming.length+' records in backup.\n';
      if(newRecords.length>0)msg+=newRecords.length+' new records will be added.\n';
      if(duplicateCount>0)msg+=duplicateCount+' already exist and will be skipped.\n';
      msg+='\nProceed?';
      if(!confirm(msg))return;
      scanHistory=scanHistory.concat(newRecords);
      scanHistory.sort(function(a,b){return b.id-a.id;});
      localStorage.setItem('deedHistory3',JSON.stringify(scanHistory));
      updateLeftPanel();
      renderHistoryPage();
      checkExportReminder();
      alert('Restore complete! '+newRecords.length+' records added.');
    }catch(err){
      alert('Could not read file: '+err.message);
    }
  };
  reader.readAsText(file);
}

  const records = scanHistory.filter(h => h.meta.county === county);

  const dataKeys=['Document Type','Tax Sale Auction Date','Recorded Date','Seller','Sale Price','Grantee','Grantee Address','Prior Owner(s)','Taxes Owed','Property Address','County','Map #','Legal Description'];
  const headers=['Scan Title','County','Status','Notes','Record Notes','Scan Date','Duplicate',...dataKeys];
  const rows=[headers,...records.map(h=>[h.meta.title||'',h.meta.county||'',h.meta.status||'No Action Yet',h.meta.notes||'',h.meta.recordNotes||'',h.meta.date||'',h.meta.isDup?'Yes':'No',...dataKeys.map(k=>(h.data[k]&&h.data[k]!=='—')?h.data[k]:'')])];
  const csv=rows.map(r=>r.map(v=>`"${String(v).replace(/"/g,'""')}"`).join(',')).join('\n');
  const a=document.createElement('a');

  a.download='taxdeedscan_'+county.toLowerCase().replace(/ /g,'_')+'_county.csv';

// ── EXPORT REMINDER ──
const REMIND_EVERY = 10;

function checkExportReminder() {
  const lastExport = parseInt(localStorage.getItem('tds_lastExportCount') || '0');
  const total = scanHistory.length;
  const unsaved = total - lastExport;
  const badge = document.getElementById('exportRemindBadge');
  if (!badge) return;
  if (total > 0 && unsaved >= REMIND_EVERY) {
    badge.style.display = 'block';
    badge.textContent = unsaved + ' new — back up!';
  } else {
    badge.style.display = 'none';
  }
  // Popup at 20 unsaved scans
  if (unsaved >= 20 && sessionStorage.getItem('tds_popupDismissed') !== '1') {
    document.getElementById('exportPopupCount').textContent = unsaved;
    document.getElementById('exportPopupOverlay').classList.add('open');
  }
}

let dbEndpoint = localStorage.getItem('tds_dbEndpoint') || '';
let dbToken = localStorage.getItem('tds_dbToken') || '';

function initDBSettings() {
  document.getElementById('dbEndpoint').value = dbEndpoint;

function saveDBSettings() {
  dbEndpoint = document.getElementById('dbEndpoint').value.trim();
  dbToken = document.getElementById('dbToken').value.trim();
  localStorage.setItem('tds_dbEndpoint', dbEndpoint);
  localStorage.setItem('tds_dbToken', dbToken);

function updateDBBadge() {
  const badge = document.getElementById('dbStatusBadge');
  if (dbEndpoint) {
    badge.textContent = 'Configured';
    badge.className = 'sync-badge pending';
  } else {
    badge.textContent = 'Off';
    badge.className = 'sync-badge off';
  }
}

async function testDBConnection() {
  const url = document.getElementById('dbEndpoint').value.trim();
  const token = document.getElementById('dbToken').value.trim();
  if (!url) { document.getElementById('dbStatusMsg').textContent = 'Please enter an endpoint URL.'; return; }
  document.getElementById('dbStatusMsg').textContent = 'Testing...';
  try {
    const headers = { 'Content-Type': 'application/json' };
    if (token) headers['Authorization'] = token.startsWith('Bearer ') ? token : 'Bearer ' + token;
    const res = await fetch(url, { method: 'OPTIONS', headers });
    document.getElementById('dbStatusMsg').textContent = '✓ Endpoint reachable (status ' + res.status + ')';
    document.getElementById('dbStatusBadge').textContent = 'Connected';
    document.getElementById('dbStatusBadge').className = 'sync-badge synced';
  } catch(e) {
    document.getElementById('dbStatusMsg').textContent = '✗ Could not reach endpoint.';
    document.getElementById('dbStatusBadge').textContent = 'Error';
    document.getElementById('dbStatusBadge').className = 'sync-badge failed';
  }
}

async function syncToDB(data, meta, entryId) {
  if (!dbEndpoint) return;
  const badge = document.getElementById('recordSyncBadge');
  if (badge) { badge.style.display = 'inline-flex'; badge.textContent = 'Syncing...'; badge.className = 'sync-badge pending'; }
  try {
    const headers = { 'Content-Type': 'application/json' };
    if (dbToken) headers['Authorization'] = dbToken.startsWith('Bearer ') ? dbToken : 'Bearer ' + dbToken;
    const payload = {
      scanId: entryId,
      scannedAt: meta.date,
      title: meta.title,
      county: meta.county,
      notes: meta.notes,
      isDuplicate: meta.isDup || false,
      documentType: data['Document Type'] !== '—' ? data['Document Type'] : null,
      taxSaleAuctionDate: data['Tax Sale Auction Date'] !== '—' ? data['Tax Sale Auction Date'] : null,
      recordedDate: data['Recorded Date'] !== '—' ? data['Recorded Date'] : null,
      seller: data['Seller'] !== '—' ? data['Seller'] : null,
      salePrice: data['Sale Price'] !== '—' ? data['Sale Price'] : null,
      grantee: data['Grantee'] !== '—' ? data['Grantee'] : null,
      granteeAddress: data['Grantee Address'] !== '—' ? data['Grantee Address'] : null,
      priorOwners: data['Prior Owner(s)'] !== '—' ? data['Prior Owner(s)'] : null,
      taxesOwed: data['Taxes Owed'] !== '—' ? data['Taxes Owed'] : null,
      propertyAddress: data['Property Address'] !== '—' ? data['Property Address'] : null,
      mapNumber: data['Map #'] !== '—' ? data['Map #'] : null,
      legalDescription: data['Legal Description'] !== '—' ? data['Legal Description'] : null,
    };
    const res = await fetch(dbEndpoint, { method: 'POST', headers, body: JSON.stringify(payload) });
    if (res.ok) {
      if (badge) { badge.textContent = '✓ Synced to DB'; badge.className = 'sync-badge synced'; }
      // Mark in local history
      const entry = scanHistory.find(h => h.id === entryId);
      if (entry) { entry.meta.synced = true; localStorage.setItem('deedHistory3', JSON.stringify(scanHistory)); }
    } else {
      if (badge) { badge.textContent = '✗ Sync failed'; badge.className = 'sync-badge failed'; }
    }
  } catch(e) {
    if (badge) { badge.textContent = '✗ Sync failed'; badge.className = 'sync-badge failed'; }
  }
}

// ── USAGE METER ──
const COST_PER_SCAN = 0.02;
let sessionScans = 0;

function initUsageMeter() {
  updateUsageDisplay();
}

function recordScanUsage() {
  sessionScans++;
  const totalScans = parseInt(localStorage.getItem('tds_totalScans') || '0') + 1;
  localStorage.setItem('tds_totalScans', totalScans);
  updateUsageDisplay();
}

function updateUsageDisplay() {
  const totalScans = parseInt(localStorage.getItem('tds_totalScans') || '0');
  const sessionCost = sessionScans * COST_PER_SCAN;
  const totalCost = totalScans * COST_PER_SCAN;
  const budgetPct = Math.min((sessionCost / 1.00) * 100, 100);

  // Header bar
  document.getElementById('usageMeterLabel').textContent = sessionScans + (sessionScans === 1 ? ' scan' : ' scans');
  document.getElementById('usageCostLabel').textContent = '~$' + sessionCost.toFixed(2);
  const fill = document.getElementById('usageBarFill');
  fill.style.width = Math.min(budgetPct, 100) + '%';
  fill.style.background = budgetPct < 50 ? '#4caf7d' : budgetPct < 80 ? '#f59e0b' : '#c0392b';

  // Popup
  document.getElementById('umpSessionScans').textContent = sessionScans;
  document.getElementById('umpTotalScans').textContent = totalScans;
  document.getElementById('umpSessionCost').textContent = '$' + sessionCost.toFixed(2);
  document.getElementById('umpTotalCost').textContent = '$' + totalCost.toFixed(2);
  document.getElementById('umpAvgCost').textContent = '$' + COST_PER_SCAN.toFixed(2);
  const barFill = document.getElementById('umpBarFill');
  barFill.style.width = Math.min(budgetPct, 100) + '%';
  barFill.style.background = budgetPct < 50 ? '#2e7d52' : budgetPct < 80 ? '#f59e0b' : '#c0392b';
}

function toggleUsagePopup() {
  const popup = document.getElementById('usagePopup');
  popup.classList.toggle('open');
  if (popup.classList.contains('open')) updateUsageDisplay();
}

function resetUsageStats() {
  if (!confirm('Reset all-time usage stats?')) return;
  localStorage.removeItem('tds_totalScans');
  updateUsageDisplay();
}

// Close popup when clicking outside
document.addEventListener('click', e => {
  const popup = document.getElementById('usagePopup');
  if (popup && popup.classList.contains('open') &&
      !popup.contains(e.target) &&
      !e.target.closest('.usage-meter')) {
    popup.classList.remove('open');
  }
});
